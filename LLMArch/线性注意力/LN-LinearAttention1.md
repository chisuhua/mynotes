━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    def flash_linear_attention(Q, K, V, kernel_fn=elu):
        # Q, K, V: [B, H, N, d]
        B, H, N, d = Q.shape
        
        # 初始化输出
        O = zeros([B, H, N, d])      # 输出 [B, H, N, d]
        
        # 分块大小
        BLOCK_Q = 64
        BLOCK_KV = 64
        
        # 外层循环: 遍历Q块
        for i in range(0, N, BLOCK_Q):          # i = 0, 64, 128, ..., 4032
            # 加载Q块到SRAM并应用核函数
            Q_i = Q[:, :, i:i+BLOCK_Q, :]       # [B, H, BLOCK_Q, d]
            phi_Q_i = kernel_fn(Q_i)            # [B, H, BLOCK_Q, d]
            
            # 初始化状态矩阵 (每个head独立)
            S_i = zeros([B, H, d, d])           # [B, H, d, d] 状态矩阵
            
            # 内层循环: 遍历K,V块，累积状态
            for j in range(0, N, BLOCK_KV):     # j = 0, 64, 128, ..., 4032
                # 加载K,V块到SRAM
                K_j = K[:, :, j:j+BLOCK_KV, :]  # [B, H, BLOCK_KV, d]
                V_j = V[:, :, j:j+BLOCK_KV, :]  # [B, H, BLOCK_KV, d]
                
                # 对K应用核函数
                phi_K_j = kernel_fn(K_j)        # [B, H, BLOCK_KV, d]
                
                # 计算状态增量并累积
                # phi_K_j^T: [B, H, d, BLOCK_KV]
                # ΔS_ij = phi_K_j^T @ V_j: [B, H, d, d]
                delta_S_ij = einsum('bhkd,bhke->bhde', phi_K_j, V_j)
                
                # 累积状态
                S_i = S_i + delta_S_ij          # [B, H, d, d]
            
            # 计算输出块
            # O_i = phi_Q_i @ S_i: [B, H, BLOCK_Q, d]
            O_i = einsum('bhd,bhde->bhe', phi_Q_i, S_i)
            
            # 存储输出
            O[:, :, i:i+BLOCK_Q, :] = O_i
        
        return O  # [B, H, N, d]

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
【8. 因果掩码（Causal Mask）处理】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    对于因果注意力（decoder-only），需要确保位置i只能看到位置≤i的信息：

    ┌──────────────────────────────────────────────────────────────────────────────────────────────────┐
    │  标准Flash Attention因果处理：                           │  Flash Linear Attention因果处理：      │
    │  - 在softmax前将未来位置设为-inf                         │  - 状态累积时只累加j≤i的块           │
    │  - 需要额外的掩码矩阵                                    │  - 内层循环j只到当前i块              │
    └──────────────────────────────────────────────────────────────────────────────────────────────────┘

    因果FLA伪代码：
    ┌──────────────────────────────────────────────────────────────────────────────────────────────────┐
    │  for i in range(0, N, BLOCK_Q):                                                                  │
    │      S_i = zeros([B, H, d, d])                                                                   │
    │      # 因果：内层循环只到当前i位置                                                                 │
    │      for j in range(0, i + BLOCK_Q, BLOCK_KV):  # j ≤ i                                         │
    │          K_j, V_j = load_KV(j)                                                                   │
    │          phi_K_j = kernel_fn(K_j)                                                                │
    │          delta_S_ij = phi_K_j^T @ V_j                                                            │
    │          S_i = S_i + delta_S_ij                                                                  │
    │      O_i = phi_Q_i @ S_i                                                                         │
    └──────────────────────────────────────────────────────────────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
【9. 递归/状态空间视角（RNN等价形式）】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    Flash Linear Attention 可以等价地表示为递归形式：

    ┌──────────────────────────────────────────────────────────────────────────────────────────────────┐
    │  递归公式：                                                                                      │
    │                                                                                                  │
    │  S_t = S_{t-1} + φ(K_t)^T @ V_t      [d, d] 状态更新                                            │
    │  O_t = φ(Q_t) @ S_t                  [d] 输出计算                                               │
    │                                                                                                  │
    │  初始状态：S_0 = 0                                                                               │
    │                                                                                                  │
    │  这等价于一个线性RNN！                                                                            │
    └──────────────────────────────────────────────────────────────────────────────────────────────────┘

    递归实现伪代码：
    ┌──────────────────────────────────────────────────────────────────────────────────────────────────┐
    │  def linear_attention_recurrent(Q, K, V, kernel_fn=elu):                                         │
    │      B, H, N, d = Q.shape                                                                        │
    │      O = zeros([B, H, N, d])                                                                     │
    │      S = zeros([B, H, d, d])  # 状态矩阵                                                         │
    │                                                                                                  │
    │      for t in range(N):  # 序列维度循环                                                          │
    │          q_t = Q[:, :, t, :]          # [B, H, d]                                                │
    │          k_t = K[:, :, t, :]          # [B, H, d]                                                │
    │          v_t = V[:, :, t, :]          # [B, H, d]                                                │
    │                                                                                                  │
    │          phi_q_t = kernel_fn(q_t)     # [B, H, d]                                                │
    │          phi_k_t = kernel_fn(k_t)     # [B, H, d]                                                │
    │                                                                                                  │
    │          # 状态更新                                                                              │
    │          S = S + einsum('bhd,bhe->bhde', phi_k_t, v_t)   # [B, H, d, d]                          │
    │                                                                                                  │
    │          # 输出计算                                                                              │
    │          O[:, :, t, :] = einsum('bhd,bhde->bhe', phi_q_t, S)  # [B, H, d]                        │
    │                                                                                                  │
    │      return O                                                                                    │
    └──────────────────────────────────────────────────────────────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
【10. 内存访问优化对比】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    ┌─────────────────────────────────────┐   ┌─────────────────────────────────────┐
    │      传统 Linear Attention          │   │      Flash Linear Attention         │
    ├─────────────────────────────────────┤   ├─────────────────────────────────────┤
    │  1. φ(Q), φ(K) 写HBM                │   │  1. φ(Q_i), φ(K_j) 只在SRAM         │
    │  2. φ(K)^T @ V → S [d,d] 写HBM      │   │  2. S 在SRAM中累积，不写HBM         │
    │  3. φ(Q) @ S → O 写HBM              │   │  3. O_i 直接写回HBM                 │
    │                                     │   │                                     │
    │  HBM访问: O(N·d²)                   │   │  HBM访问: O(N·d)                    │
    │  中间矩阵: [d,d] 可能需要存储       │   │  中间矩阵: S在SRAM累积              │
    │  显存: O(N·d + d²)                  │   │  显存: O(N·d)                       │
    └─────────────────────────────────────┘   └─────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
【11. 关键维度总结】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    输入/输出维度:  [B, H, N, d]
                    │  │  │  └─ 头维度 (head dimension)，如64/128/256
                    │  │  └──── 序列长度 (sequence length)
                    │  └─────── 注意力头数 (num attention heads)
                    └────────── 批量大小 (batch size)
    
    核函数映射维度:  φ(Q), φ(K): [B, H, N, d]  ← 维度不变，值域变换
    
    块内计算维度:   φ(Q_i): [BLOCK_Q, d]
                    φ(K_j): [BLOCK_KV, d]
                    ΔS_ij: [d, d]  ← 状态增量矩阵（关键！）
                    S_i: [d, d]  ← 累积状态矩阵
                    O_i: [BLOCK_Q, d]  ← 输出块
    
    与Flash Attention对比:
    ┌──────────────────────────────────────────────────────────────────────────────────────────────────┐
    │  维度              │  Flash Attention    │  Flash Linear Attention                               │
    ├──────────────────────────────────────────────────────────────────────────────────────────────────┤
    │  注意力矩阵        │  [N, N] 二次方       │  不显式计算                                          │
    │  中间状态          │  [N, N] 需要存储     │  [d, d] SRAM中累积                                   │
    │  内存复杂度        │  O(N²)              │  O(N)                                                 │
    │  计算复杂度        │  O(N²·d)            │  O(N·d²)                                              │
    │  适合场景          │  短序列，高精度      │  长序列，高效推理                                     │
    └──────────────────────────────────────────────────────────────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
【12. Triton内核实现要点】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    @triton.jit
    def flash_linear_attention_kernel(
        Q_ptr, K_ptr, V_ptr, O_ptr,
        stride_qb, stride_qh, stride_qn, stride_qd,
        stride_kb, stride_kh, stride_kn, stride_kd,
        stride_vb, stride_vh, stride_vn, stride_vd,
        stride_ob, stride_oh, stride_on, stride_od,
        B, H, N, d,
        BLOCK_Q: tl.constexpr,
        BLOCK_KV: tl.constexpr,
    ):
        # 程序ID定位到具体的(batch, head, Q块)
        pid_b = tl.program_id(0)
        pid_h = tl.program_id(1)
        pid_i = tl.program_id(2)
        
        # 初始化状态矩阵 S [d, d]
        S = tl.zeros([d, d], dtype=tl.float32)
        
        # 外层：遍历Q块
        q_offset = pid_b * stride_qb + pid_h * stride_qh + pid_i * BLOCK_Q * stride_qn
        Q_i = tl.load(Q_ptr + q_offset + tl.arange(0, BLOCK_Q)[:, None] * stride_qn + 
                      tl.arange(0, d)[None, :] * stride_qd)
        phi_Q_i = kernel_fn(Q_i)  # 核函数映射
        
        # 内层：遍历K,V块，累积状态
        for j in range(0, N, BLOCK_KV):
            k_offset = pid_b * stride_kb + pid_h * stride_kh + j * stride_kn
            v_offset = pid_b * stride_vb + pid_h * stride_vh + j * stride_vn
            
            K_j = tl.load(K_ptr + k_offset + ...)
            V_j = tl.load(V_ptr + v_offset + ...)
            
            phi_K_j = kernel_fn(K_j)
            
            # 状态增量：phi_K_j^T @ V_j
            delta_S = tl.dot(phi_K_j.T, V_j)
            S += delta_S
        
        # 输出：phi_Q_i @ S
        O_i = tl.dot(phi_Q_i, S)
        
        # 写回HBM
        o_offset = pid_b * stride_ob + pid_h * stride_oh + pid_i * BLOCK_Q * stride_on
        tl.store(O_ptr + o_offset + ..., O_i)
