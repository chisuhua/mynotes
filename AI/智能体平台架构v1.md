 1. 执行摘要
本文档定义了基于C++构建的本地私有化AI编码助手平台完整架构。该平台作为llama.cpp的智能体编排层，通过浏览器提供类ChatGPT的流式对话与原生终端（xterm.js）混合交互体验。
核心架构决策：采用双端口分离架构而非单一框架方案，以在资源受限环境（与4-8GB模型推理共存）下实现最佳性能隔离与开发效率平衡。

 2. 架构愿景与约束
 2.1 业务目标
- 本地私有化：所有数据（模型、代码、文件）留存本地工作站
- 混合交互：自然语言对话（AI助手）+ 命令行终端（xterm.js）无缝切换
- 多Session并发：支持3-5个独立工作会话，Session间状态完全隔离
- 资源敏感：HTTP服务内存占用<50MB，避免与llama.cpp抢占物理内存
 2.2 技术约束矩阵

延迟敏感	xterm.js输入延迟<10ms（本地体验要求）	µWebSockets独立进程，绑定独立CPU核心
安全隔离	AI生成的代码需在沙箱执行	Docker + seccomp多层沙箱
开发效率	团队熟悉C++17/20，拒绝造轮子	Cinatra（HTTP）+ µWebSockets（WS）成熟框架

// µWebSockets终端服务伪代码
#include <uwebsockets/App.h>

struct TerminalSession {
    uWS::WebSocket<false, true>* ws;
    std::unique_ptr<DockerPTY> pty;
    std::string session_id;
};

void startTerminalServer() {
    uWS::App().ws<PerSocketData>("/*", {
        .open = [](auto *ws) {
            std::string path = ws->getUrl();
            std::string session_id = extractSessionId(path); // /terminal/{id}
            
            // 验证Session存在（查询Cinatra或共享内存）
            if (!validateSession(session_id)) {
                ws->close();
                return;
            }
            
            // 启动Docker沙箱
            auto pty = std::make_unique<DockerPTY>(session_id);
            ws->getUserData()->terminal = std::move(pty);
            ws->getUserData()->session_id = session_id;
        },
        .message = [](auto *ws, std::string_view message, uWS::OpCode opCode) {
            auto* data = ws->getUserData();
            // 解析JSON命令
            json cmd = json::parse(message);
            if (cmd["type"] == "input") {
                data->terminal->write(cmd["data"]); // 写入PTY
            } else if (cmd["type"] == "resize") {
                data->terminal->resize(cmd["cols"], cmd["rows"]); // 终端大小调整
            }
        },
        .close = [](auto *ws, int code, std::string_view message) {
            auto* data = ws->getUserData();
            data->terminal->shutdown(); // 清理Docker容器
        }
    }).listen(8081, [](auto *token) {
        if (token) std::cout << "Terminal server listening on port 8081\n";
    }).run();
}

// 前端动态预览路由
function previewFile(sessionId, filePath) {
    const ext = filePath.split('.').pop().toLowerCase();
    const apiBase = `http://localhost:8080/api/session/${sessionId}/file`;
    
    switch(ext) {
        case 'md':
            fetch(`${apiBase}?path=${filePath}`)
                .then(r => r.text())
                .then(md => {
                    const html = marked.parse(md);
                    injectHighlightJs(); // 代码高亮
                    showPreview(html, 'markdown');
                });
            break;
            
        case 'pdf':
            showPreview('', 'pdf');
            loadPDF(`${apiBase}?path=${filePath}`); // pdf.js渲染
            break;
            
        case 'jpg': case 'png': case 'gif':
            showPreview(`${apiBase}?path=${filePath}`, 'image');
            break;
            
        case 'cpp': case 'py': case 'js':
            fetch(`${apiBase}?path=${filePath}`)
                .then(r => r.text())
                .then(code => {
                    monacoEditor.setValue(code);
                    monacoEditor.setModelLanguage(ext);
                    switchTab('code');
                });
            break;
            
        default:
            // 纯文本
            fetch(`${apiBase}?path=${filePath}`)
                .then(r => r.text())
                .then(text => showPreview(`<pre>${escapeHtml(text)}</pre>`, 'text'));
    }
}

9. 命令注入防护：终端输入经过严格白名单过滤，禁止shell元字符
10. 资源限制：Docker容器 --pids-limit 50 --memory 512m 防止fork炸弹
11. 本地绑定：服务默认绑定127.0.0.1，拒绝远程连接（除非显式配置）

 8. 部署与运维架构
 8.1 进程与资源分配
#!/bin/bash
# start.sh - 生产环境启动脚本

# 1. 检查端口占用
check_port() {
    if lsof -Pi :$1 -sTCP:LISTEN -t >/dev/null ; then
        echo "Error: Port $1 is already in use"
        exit 1
    fi
}
check_port 8080 && check_port 8081 && check_port 8082

# 2. 启动llama.cpp（绑定CPU 2-3，避免阻塞终端）
echo "Starting llama.cpp server..."
taskset -c 2,3 ./llama-server \
    -m ./models/coder-7b.gguf \
    --port 8082 \
    --ctx-size 4096 \
    -t 4 &

LLAMA_PID=$!
sleep 5  # 等待模型加载

# 3. 启动Cinatra HTTP服务（绑定CPU 0）
echo "Starting Cinatra HTTP service..."
taskset -c 0 ./cinatra_server \
    --port 8080 \
    --workspace ./workspace \
    --llama-url http://localhost:8082 &

CINATRA_PID=$!

# 4. 启动µWebSockets终端服务（绑定CPU 1，确保响应）
echo "Starting µWebSockets terminal service..."
taskset -c 1 ./uws_terminal \
    --port 8081 \
    --docker-image coding-agent-sandbox \
    --workspace ./workspace &

UWS_PID=$!

# 5. 健康检查
echo "Checking health..."
sleep 2
curl -s http://localhost:8080/health || echo "Warning: Cinatra not responding"
curl -s http://localhost:8081/health || echo "Warning: µWebSockets not responding"

echo "All services started!"
echo "Cinatra PID: $CINATRA_PID"
echo "UWS PID: $UWS_PID"
echo "Llama PID: $LLAMA_PID"

# 6. 优雅关闭
trap "kill $LLAMA_PID $CINATRA_PID $UWS_PID; exit" SIGINT SIGTERM
wait

 11. 风险与缓解策略
风险	可能性	影响	缓解措施
µWebSockets与Cinatra端口冲突	中	高	启动脚本增加端口占用检测，支持动态端口分配（find_free_port()）
llama.cpp推理阻塞HTTP服务	高	高	绑定不同CPU核心（taskset）；Cinatra使用独立线程池；流式响应异步处理
Docker沙箱逃逸	低	极高	多层安全（seccomp+user namespaces+cap drop）；定期更新基础镜像；禁止privileged模式
Session状态不一致	中	中	前端实现重连机制；定期同步Session列表；异常时自动刷新
大文件上传内存溢出	中	高	Cinatra使用流式multipart解析，写入临时文件而非内存；设置文件大小限制（<100MB）
浏览器兼容性问题	低	中	前端支持Chrome/Firefox/Safari最新两个主版本；WebSocket降级提示

 12.1 术语表
