浏览器智能体融合架构（Browser Agent Fusion Architecture, BAFA）
版本：v1.0
适用场景：浏览器自动化、Web UI 智能体、多标签页任务执行
核心特性：硬实时中断 + 协程化任务 + 结构化持久化

 1.2 关键融合决策
层级	采用方案	融合来源	解决痛点
Layer 1	无锁中断环 + 优先级通道	ZeroClaw 事件总线	浏览器崩溃、用户暂停、验证码弹窗的硬实时响应
Layer 2	C++20 协程 + 显式状态检查点	用户的事件驱动 ReAct	避免回调地狱，同时支持跨页面导航的状态持久化
Layer 3	独立监督线程 + 目标漂移检测	分层认知架构	防止任务漂移（如从"比价"变成"浏览"）

 2. Layer 1: 反射层（硬实时）
设计目标：<1ms 延迟响应安全关键事件，永不阻塞
 2.1 架构组件
// layer1/interrupt_controller.h
#include <atomic>
#include <ring_buffer.hpp>

enum class InterruptPriority : uint8_t {
    CRITICAL = 0,   // 浏览器崩溃、OOM、安全警报
    USER = 1,       // 暂停、停止、输入干预
    SYSTEM = 2,     // 页面卸载(beforeunload)、弹窗
    BACKGROUND = 3  // 网络状态变化
};

struct InterruptEvent {
    InterruptPriority priority;
    uint64_t timestamp_us;
    std::variant<
        UserInterrupt,      // 用户指令
        BrowserAlert,       // 验证码、登录框
        PageLifecycleEvent, // load/unload/crash
        SecurityAlert       // XSS、钓鱼检测
    > payload;
    
    // 无锁队列所需：原子指针
    std::atomic<InterruptEvent*> next{nullptr};
};

class InterruptController {
private:
    // 无锁环形缓冲区（每核一个，避免伪共享）
    alignas(64) RingBuffer<InterruptEvent, 1024> urgent_queue_;
    alignas(64) RingBuffer<InterruptEvent, 4096> normal_queue_;
    
    std::atomic<bool> freeze_signal_{false};
    std::atomic<uint64_t> frozen_timestamp_{0};
    
public:
    // 非阻塞提交，CPU 自旋锁（<100ns 延迟）
    bool PostInterrupt(InterruptEvent event, InterruptPriority pri);
    
    // Layer 2 定期轮询检查点（协作式抢占）
    bool CheckFreezeSignal();
    
    // 紧急制动：向浏览器进程发送 SIGSTOP
    void EmergencyFreeze();
    
    // 恢复：从 SQLite 读取状态，重建 Layer 2 协程
    void ResumeFromCheckpoint();
};

// Layer 2 的协程必须定期检查点
struct TaskCoroutine {
    std::coroutine_handle<> handle;
    TaskContext* context;
    
    // 每个 await 点插入检查
    bool CheckInterrupt() {
        if (interrupt_controller_.CheckFreezeSignal()) {
            // 1. 保存当前状态到 SQLite
            SaveCheckpoint();
            // 2. 挂起协程（但不销毁）
            handle.suspend();
            return true;
        }
        return false;
    }
};

 3. Layer 2: 认知层（软实时）
设计目标：执行 ReAct 循环，支持长时间任务（小时级），可跨页面导航恢复
 3.1 协程化 ReAct 引擎
// layer2/cognitive_engine.h
#include <coroutine>
#include <optional>

// 显式状态枚举（用于持久化）
enum class TaskState {
    PLANNING,
    ACTING,
    OBSERVING,
    REFLECTING,
    WAITING_USER_INPUT,
    RECOVERING,
    COMPLETED,
    FAILED
};

struct ReActStep {
    std::string thought;
    Action action;
    Observation observation;
    std::chrono::steady_clock::time_point timestamp;
};

// 可持久化的上下文（JSON 序列化）
struct PersistentContext {
    std::string task_id;
    std::string goal;
    TaskState current_state;
    std::vector<ReActStep> history;
    nlohmann::json browser_state;  // 页面 URL、视口、表单填充状态
    uint32_t retry_count;
};

class CognitiveEngine {
public:
    // 主入口：恢复现有任务或创建新任务
    Task<void> ExecuteTask(std::optional<std::string> resume_from_id);
    
private:
    // 协程化的 ReAct 步骤（看起来像同步，实际是异步事件驱动）
    Task<ReActStep> PlanningStep(PersistentContext& ctx);
    Task<Observation> ActingStep(const Action& action, PersistentContext& ctx);
    Task<ReflectionResult> ReflectionStep(const Observation& obs, PersistentContext& ctx);
    
    // 关键：每个步骤都持久化
    Task<void> SaveCheckpoint(const PersistentContext& ctx) {
        co_await db_->Write(ctx.task_id, ctx.to_json());
    }
    
    // 等待 Layer 1 的恢复信号
    Task<void> WaitForResume();
};

// 协程实现示例
Task<void> CognitiveEngine::ExecuteTask(std::optional<std::string> resume_id) {
    // 恢复或创建上下文
    PersistentContext ctx = resume_id 
        ? co_await LoadContext(*resume_id) 
        : CreateNewContext();
    
    while (ctx.current_state != TaskState::COMPLETED && 
           ctx.current_state != TaskState::FAILED) {
        
        // 检查点 1：可被 Layer 1 中断
        if (co_await CheckInterrupt()) {
            co_await WaitForResume();
            continue;
        }
        
        co_await SaveCheckpoint(ctx);
        
        switch (ctx.current_state) {
            case TaskState::PLANNING: {
                auto step = co_await PlanningStep(ctx);
                ctx.history.push_back(step);
                ctx.current_state = TaskState::ACTING;
                break;
            }
                
            case TaskState::ACTING: {
                // 执行浏览器操作（异步，不阻塞线程）
                auto obs = co_await ActingStep(ctx.history.back().action, ctx);
                
                // 检查点 2：页面导航后保存状态
                co_await SaveCheckpoint(ctx);
                
                ctx.history.back().observation = obs;
                ctx.current_state = TaskState::OBSERVING;
                break;
            }
                
            case TaskState::OBSERVING: {
                // 并行观察：截图 + DOM 提取 + 网络状态
                auto [screenshot, dom, network] = co_await std::when_all(
                    browser_->Screenshot(),
                    browser_->ExtractDOM(),
                    browser_->GetNetworkStatus()
                );
                
                auto reflection = co_await ReflectionStep(
                    Observation{screenshot, dom, network}, 
                    ctx
                );
                
                if (reflection.is_complete) {
                    ctx.current_state = TaskState::COMPLETED;
                } else if (reflection.needs_user_input) {
                    ctx.current_state = TaskState::WAITING_USER_INPUT;
                    // 通知 Layer 1 进入等待模式
                    co_await NotifyLayer1Wait();
                } else {
                    ctx.current_state = TaskState::PLANNING;
                }
                break;
            }
                
            case TaskState::WAITING_USER_INPUT: {
                // 挂起协程，等待 Layer 1 的 USER_INPUT 事件
                auto user_input = co_await WaitForUserInterrupt();
                ctx.history.push_back({
                    .thought = "Received user input: " + user_input,
                    .action = Action{ActionType::NONE},
                    .observation = {}
                });
                ctx.current_state = TaskState::PLANNING;
                break;
            }
                
            case TaskState::RECOVERING: {
                // 从崩溃恢复：重新连接浏览器，恢复页面状态
                co_await RecoverBrowserState(ctx.browser_state);
                ctx.current_state = TaskState::PLANNING;
                break;
            }
        }
    }
    
    co_await SaveCheckpoint(ctx);  // 最终状态
    co_await ReportResult(ctx);
}

 3.3 多标签页并发模型
每个标签页拥有 独立的 Layer 2 实例，共享 Layer 1 和 Layer 3：
class TabManager {
    std::unordered_map<std::string, std::unique_ptr<CognitiveEngine>> tabs_;
    
public:
    // 为新标签页创建独立协程
    void SpawnTab(const std::string& tab_id, const std::string& goal) {
        auto engine = std::make_unique<CognitiveEngine>(tab_id, goal);
        
        // 每个 Tab 在自己的协程调度器上运行
        auto task = engine->ExecuteTask(std::nullopt);
        
        tabs_[tab_id] = std::move(engine);
    }
    
    // Layer 1 可单独暂停某个 Tab
    void FreezeTab(const std::string& tab_id) {
        tabs_[tab_id]->RequestFreeze();
    }
    
    // Layer 3 可全局协调：当检测到多标签页冲突时暂停次要任务
    void CoordinateTabs(const GoalAnalysis& analysis) {
        for (auto& [id, engine] : tabs_) {
            if (analysis.IsSecondary(id)) {
                engine->SetPriority(TaskPriority::BACKGROUND);
            }
        }
    }
};

设计目标：监督 Layer 2，防止目标漂移，优化资源使用
 4.1 目标漂移检测
// layer3/metacognition.h
class GoalMonitor {
    // 每 30 秒或每完成 5 个 ReAct 步骤运行一次
    void PeriodicCheck(const std::string& task_id) {
        auto ctx = db_->LoadContext(task_id);
        
        // 1. 提取当前行动轨迹
        auto trajectory = ExtractTrajectory(ctx.history);
        
        // 2. 与原始目标对比（使用 LLM 或规则引擎）
        auto drift_score = CalculateGoalDrift(ctx.goal, trajectory);
        
        if (drift_score > 0.7) {  // 严重漂移
            // 向 Layer 2 发送策略调整信号
            layer2_->InjectReflection(
                "Warning: You seem to be drifting from the original goal. "
                "Current trajectory focuses on browsing rather than purchasing. "
                "Consider refocusing or ask user for clarification."
            );
        }
        
        // 3. 检测循环（Repetitive loops）
        if (DetectRepetitivePattern(ctx.history, 3)) {  // 重复 3 次以上
            layer2_->RequestUserIntervention(
                "Stuck in a loop. Please provide guidance."
            );
        }
    }
};

class ContextCompaction {
    Task<void> CompactHistory(PersistentContext& ctx) {
        if (ctx.history.size() < 50) co_return;
        
        // 保留关键步骤（错误、用户干预、里程碑）
        auto critical_steps = FilterCriticalSteps(ctx.history);
        
        // 其余步骤使用 LLM 生成摘要
        auto summary = co_await llm_->SummarizeSteps(
            std::vector<ReActStep>(
                ctx.history.begin() + 10, 
                ctx.history.end() - 5
            )
        );
        
        ctx.history = critical_steps;
        ctx.history.push_back({
            .thought = "Summary of previous actions: " + summary,
            .action = Action{ActionType::SUMMARY_MARKER},
            .observation = {}
        });
        
        co_await db_->Write(ctx.task_id, ctx.to_json());
    }
};

 5.2 崩溃恢复流程
Task<void> RecoveryManager::RecoverFromCrash() {
    // 1. 查询未完成的任务
    auto interrupted_tasks = co_await db_->Query(
        "SELECT * FROM tasks WHERE current_state NOT IN ('COMPLETED', 'FAILED')"
    );
    
    for (auto& task : interrupted_tasks) {
        // 2. 检查浏览器状态
        if (task.browser_snapshot.is_null()) {
            // 无快照，无法恢复，标记失败
            co_await db_->UpdateStatus(task.task_id, "FAILED");
            continue;
        }
        
        // 3. 重新启动浏览器实例
        auto new_browser = co_await browser_factory_->Launch();
        
        // 4. 恢复页面状态（Cookie、LocalStorage、URL）
        co_await new_browser->RestoreSnapshot(
            BrowserSnapshot::FromJson(task.browser_snapshot)
        );
        
        // 5. 重建 Layer 2 协程，从断点继续
        auto engine = std::make_unique<CognitiveEngine>(new_browser);
        spawn(engine->ExecuteTask(task.task_id));  // 传入 task_id 触发恢复逻辑
    }
}

 8. 与 ZeroClaw 的集成方案
如需将此架构集成到 ZeroClaw（Rust 实现）：
// 在 ZeroClaw 中创建 BrowserAgent Channel
pub struct BrowserChannel {
    // Layer 1: 使用 Rust 的 crossbeam 无锁通道
    interrupt_tx: crossbeam::channel::Sender<InterruptEvent>,
    
    // Layer 2: 使用 tokio::task 模拟协程（或引入 async_recursion）
    task_handles: Arc<Mutex<HashMap<String, JoinHandle<()>>>>,
    
    // Layer 3: 独立线程运行监控
    monitor_handle: Option<JoinHandle<()>>,
}

#[async_trait]
impl Channel for BrowserChannel {
    async fn start(&self) -> Result<()> {
        // 启动 Layer 1 监听（Playwright CDP）
        spawn_cdp_listener(self.interrupt_tx.clone());
        
        // 启动 Layer 3 监控
        self.monitor_handle = Some(tokio::spawn(metacognition_loop()));
        
        Ok(())
    }
    
    async fn send(&self, msg: Message) -> Result<()> {
        // 用户消息进入 Layer 2 协程
        let handle = tokio::spawn(async move {
            let engine = CognitiveEngine::new();
            engine.execute(msg.content).await
        });
        
        self.task_handles.lock().await.insert(msg.id, handle);
        Ok(())
    }
    
    // Layer 1 中断处理
    async fn handle_interrupt(&self, event: InterruptEvent) {
        match event.priority {
            InterruptPriority::CRITICAL | InterruptPriority::USER => {
                // 暂停所有任务
                for (id, handle) in self.task_handles.lock().await.iter() {
                    handle.abort(); // 或发送暂停信号
                    save_checkpoint(id).await;
                }
            }
            _ => {}
        }
    }
}
