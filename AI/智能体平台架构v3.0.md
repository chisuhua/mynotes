 B. Host Functions（宿主函数）
WASM 模块无法直接访问系统资源，必须通过 Host Functions 进行受控访问：
// host_functions.cpp - 提供给 WASM 的接口实现

#include <extism.hpp>

// 1. LLM 子调用（让 WASM 技能也能使用 AI）
extism::Val agent_llm_call(std::vector<extism::Val>& inputs, void* user_data) {
    // inputs[0]: prompt (string ptr)
    // inputs[1]: max_tokens (i32)
    auto* context = static_cast<AgentContext*>(user_data);
    std::string prompt = GetStringInput(inputs[0]);
    
    // 调用 llama.cpp（复用已有连接）
    std::string response = context->llm_client->QuickCompletion(prompt, inputs[1].i32);
    
    // 返回字符串（分配内存给 WASM）
    return AllocStringOutput(response);
}

// 2. 虚拟文件系统（Session 隔离）
extism::Val vfs_read(std::vector<extism::Val>& inputs, void* user_data) {
    auto* context = static_cast<AgentContext*>(user_data);
    std::string path = GetStringInput(inputs[0]);
    
    // 安全路径检查（必须在当前 Session 工作区内）
    std::string safe_path = ValidateAndResolvePath(context->session_id, path);
    if (safe_path.empty()) {
        return ErrorOutput("Permission denied: Path outside workspace");
    }
    
    std::string content = ReadFile(safe_path);
    return AllocStringOutput(content);
}

// 3. 受控 HTTP 请求（WASM 可以安全地访问外部 API）
extism::Val http_request(std::vector<extism::Val>& inputs, void* user_data) {
    auto* context = static_cast<AgentContext*>(user_data);
    json request = json::parse(GetStringInput(inputs[0]));
    
    // 检查白名单（只允许特定域名，如 api.github.com）
    if (!IsDomainWhitelisted(request["url"])) {
        return ErrorOutput("Domain not in whitelist");
    }
    
    // 执行请求（通过 libcurl）
    auto response = HttpClient::Request(request);
    return AllocStringOutput(response.dump());
}

// 注册所有 Host Functions
void RegisterHostFunctions(extism::Plugin& plugin, AgentContext* context) {
    plugin.set_user_data(context);
    
    plugin.add_function("agent_llm_call", 
                       {extism::ValType::I64, extism::ValType::I32},  // 输入: ptr, max_tokens
                       {extism::ValType::I64},                        // 输出: ptr
                       agent_llm_call);
    
    plugin.add_function("vfs_read", 
                       {extism::ValType::I64}, 
                       {extism::ValType::I64}, 
                       vfs_read);
    
    plugin.add_function("vfs_write", 
                       {extism::ValType::I64, extism::ValType::I64},  // path, content
                       {extism::ValType::I32},                        // 成功标志
                       vfs_write);
    
    plugin.add_function("http_request",
                       {extism::ValType::I64},
                       {extism::ValType::I64},
                       http_request);
    
    plugin.add_function("agent_log",
                       {extism::ValType::I64, extism::ValType::I32},  // message, level
                       {},
                       agent_log);
}

 C. WASM Skill 实例（Rust 示例）
// rust_formatter_skill.rs - 编译为 wasm32-unknown-unknown
// 使用 extism-pdk (Plugin Development Kit)

use extism_pdk::*;

// 定义输入输出结构（会自动序列化为 JSON）
#[derive(FromJson)]
struct Input {
    code: String,
    max_width: Option<i32>,
}

#[derive(ToJson)]
struct Output {
    formatted_code: String,
    errors: Vec<String>,
    line_count: i32,
}

// 主机函数声明（由 C++ 后端提供）
extern "C" {
    fn agent_llm_call(ptr: i64, max_tokens: i32) -> i64;
    fn vfs_write(path: i64, content: i64) -> i32;
    fn agent_log(msg: i64, level: i32);
}

// 主入口函数（被 C++ 后端调用）
#[plugin_fn]
pub fn format_code(input: Json<Input>) -> FnResult<Json<Output>> {
    // 1. 记录日志
    unsafe { 
        agent_log("Starting Rust format".as_ptr() as i64, 1); // level 1 = INFO
    }
    
    // 2. 执行格式逻辑（纯 Rust，无需外部依赖）
    let formatted = match rustfmt_lib::format_code(&input.code, input.max_width) {
        Ok(code) => code,
        Err(e) => {
            return Ok(Json(Output {
                formatted_code: input.code.clone(),
                errors: vec![e.to_string()],
                line_count: 0,
            }));
        }
    };
    
    // 3. 可选：调用 LLM 进行进一步优化（子调用）
    let prompt = format!("Review this Rust code for idiomatic improvements:\n{}", formatted);
    let llm_ptr = unsafe { agent_llm_call(prompt.as_ptr() as i64, 200) };
    let suggestions = ptr_to_string(llm_ptr);
    
    // 4. 可选：写回文件系统（通过 Host Function）
    if suggestions.contains("safety") {
        unsafe {
            vfs_write("/tmp/safety_notes.txt".as_ptr() as i64, 
                     suggestions.as_ptr() as i64);
        }
    }
    
    let lines = formatted.lines().count() as i32;
    
    Ok(Json(Output {
        formatted_code: formatted,
        errors: vec![],
        line_count: lines,
    }))
}

 3.1 Skill 作为 First-Class Action
Agent 现在可以选择执行 Native Tool 或 WASM Skill：
// agent.cpp - 决策逻辑更新

void Agent::DecisionStep(const std::string& llm_output) {
    json decision = json::parse(llm_output);
    
    if (decision.contains("tool")) {
        // 原生工具（Docker/系统级）
        ExecuteNativeTool(decision["tool"], decision["params"]);
    } 
    else if (decision.contains("skill")) {
        // WASM 技能（算法/轻量操作）
        ExecuteWasmSkill(decision["skill"], decision["input"]);
    }
}

void Agent::ExecuteWasmSkill(const std::string& skill_name, const json& input) {
    // 1. 记录到执行轨迹
    AddToTrace("action", {{"type": "skill", "name": skill_name, "input": input}});
    
    // 2. 通过 Skill Registry 执行（带超时）
    auto future = std::async(std::launch::async, [&]() {
        return SkillRegistry::Instance().ExecuteSkill(skill_name, input, session_id_);
    });
    
    // 3. 等待结果（可取消）
    if (future.wait_for(std::chrono::seconds(10)) == std::future_status::timeout) {
        ObservationStep({{"error": "Skill execution timeout (>10s)"}});
        return;
    }
    
    json result = future.get();
    
    // 4. 观察结果
    ObservationStep(result);
    
    // 5. 通知前端（Skill 执行完成）
    BroadcastUpdate({
        {"type", "skill_complete"},
        {"skill", skill_name},
        {"duration_ms", timer.elapsed()},
        {"result_size", result.dump().size()}
    });
}

bool SkillRegistry::ValidatePermission(const std::string& skill_name, 
                                      const std::string& capability) {
    auto& manifest = manifests_[skill_name];
    
    // 显式禁止检查
    if (std::any_of(manifest.prohibited.begin(), manifest.prohibited.end(),
                   [&](const auto& p) { return MatchWildcard(capability, p); })) {
        return false;
    }
    
    // 显式允许检查
    return std::any_of(manifest.permissions.begin(), manifest.permissions.end(),
                      [&](const auto& p) { return MatchWildcard(capability, p); });
}

 5. 性能优化策略
 5.1 WASM 实例池（避免冷启动）
class WasmInstancePool {
private:
    std::unordered_map<std::string, 
        std::deque<std::shared_ptr<extism::Plugin>>> available_;
    std::mutex mutex_;
    
public:
    // 预热：启动时创建 N 个实例
    void Preheat(const std::string& skill_name, int count) {
        for (int i = 0; i < count; ++i) {
            auto plugin = CreateFreshInstance(skill_name);
            available_[skill_name].push_back(plugin);
        }
    }
    
    // 获取实例（无锁快速路径）
    std::shared_ptr<extism::Plugin> Acquire(const std::string& skill_name) {
        std::lock_guard lock(mutex_);
        auto& queue = available_[skill_name];
        if (!queue.empty()) {
            auto plugin = queue.front();
            queue.pop_front();
            return plugin;
        }
        // 池耗尽时创建新实例（慢路径）
        return CreateFreshInstance(skill_name);
    }
    
    // 归还实例（重置内存后复用）
    void Release(const std::string& skill_name, std::shared_ptr<extism::Plugin> plugin) {
        plugin->reset_memory();  // 清理上次调用的残留数据
        std::lock_guard lock(mutex_);
        available_[skill_name].push_back(plugin);
    }
};

 5.2 异步执行与流水线
// Agent 可以同时调用多个独立的 WASM Skills
void Agent::ParallelSkillExecution(const std::vector<std::string>& skills, 
                                  const json& input) {
    std::vector<std::future<json>> futures;
    
    for (const auto& skill : skills) {
        futures.push_back(
            std::async(std::launch::async, [&]() {
                return SkillRegistry::Instance().ExecuteSkill(skill, input, session_id_);
            })
        );
    }
    
    // 收集所有结果
    std::vector<json> results;
    for (auto& f : futures) {
        results.push_back(f.get());
    }
    
    // 合并观察结果
    ObservationStep(MergeResults(results));
}

svr.post("/api/session/:id/skills/:name/execute", [](const httplib::Request& req, httplib::Response& res) {
    auto session_id = req.path_params.at("id");
    auto skill_name = req.path_params.at("name");
    auto input = json::parse(req.body);
    
    // 权限检查：Session 是否有权执行此 Skill
    if (!CheckSessionPermission(session_id, skill_name)) {
        res.status = 403;
        res.set_content(R"({"error": "Skill not permitted for this session"})", "application/json");
        return;
    }
    
    // 执行 WASM Skill
    auto result = SkillRegistry::Instance().ExecuteSkill(skill_name, input, session_id);
    res.set_content(result.dump(), "application/json");
});
