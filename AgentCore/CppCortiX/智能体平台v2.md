 4.3 工具执行流程 (ReAct 中的 Action)
// Agent 执行工具的完整流程
void Agent::ActionStep(const json& action) {
    state_ = AgentState::ACTION;
    
    std::string tool_name = action["tool"];
    json params = action["parameters"];
    
    // 1. 通过 ToolRegistry 获取工具实例
    auto tool = ToolRegistry::Instance().GetTool(tool_name);
    
    // 2. 执行前权限检查（Session隔离）
    if (!tool->ValidatePermissions(session_id_, params)) {
        ObservationStep({{"error", "Permission denied: 不能访问其他Session目录"}});
        return;
    }
    
    // 3. 异步执行工具（防止阻塞）
    tool->ExecuteAsync(params, [this](const json& result) {
        // 4. 观察结果处理
        ObservationStep(result);
        
        // 5. 继续 ReAct 循环（除非已完成）
        if (state_ != AgentState::FINAL_ANSWER && iteration_count_ < config_.max_iterations) {
            RunReActStep();
        }
    });
    
    // 6. 通知前端工具正在执行（实时状态更新）
    AgentManager::Instance().BroadcastAgentState(
        config_.agent_id, 
        {{"state", "ACTION"}, {"tool", tool_name}, {"params", params}}
    );
}

class CoordinatorAgent : public Agent {
    void ProcessComplexTask(const std::string& task) {
        // 1. 分析任务并拆解
        auto subtasks = DecomposeTask(task); // 调用LLM拆解
        
        // 2. 创建子Agent（或复用已有）
        std::vector<std::shared_ptr<Agent>> workers;
        for (const auto& sub : subtasks) {
            auto worker = AgentManager::Instance().CreateAgent(
                sub.agent_type, 
                session_id_ + "_sub_" + std::to_string(sub.id)
            );
            workers.push_back(worker);
        }
        
        // 3. 并行执行子任务
        std::vector<std::future<json>> futures;
        for (size_t i = 0; i < subtasks.size(); ++i) {
            futures.push_back(
                std::async(std::launch::async, [&, i]() {
                    return workers[i]->Execute(subtasks[i].description);
                })
            );
        }
        
        // 4. 收集结果并综合
        std::vector<json> results;
        for (auto& f : futures) results.push_back(f.get());
        
        // 5. 生成最终报告
        return SynthesizeResults(results);
    }
};

void Agent::BuildContext() {
    // 1. 当前任务注入
    std::string context = "Current Task: " + current_task_ + "\n\n";
    
    // 2. 工作记忆（最近的交互）
    context += "Recent History:\n";
    for (const auto& msg : short_term_memory_) {
        context += msg["role"].get<std::string>() + ": " + 
                   msg["content"].get<std::string>() + "\n";
    }
    
    // 3. 长期记忆检索（如果可用）
    if (long_term_memory_) {
        auto relevant = long_term_memory_->Search(
            current_task_, 
            3  // 取最相关的3条
        );
        if (!relevant.empty()) {
            context += "\nRelevant Context from Past:\n";
            for (const auto& mem : relevant) {
                context += "- " + mem.content + "\n";
            }
        }
    }
    
    // 4. 工具描述注入
    context += "\nAvailable Tools:\n" + BuildToolDescriptions();
    
    return context;
}

- - [ ] Agent Core：实现基础Agent类、ReAct循环、状态机
- - [ ] Tool Framework：定义Tool基类、Registry、Schema生成
- - [ ] 基础Tools：FileTool（读/写）、TerminalTool（沙箱执行）
- - [ ] Agent Manager：生命周期管理、Session绑定
- - [ ] CoderAgent：首个专业Agent，支持简单代码问答
 Phase 2: 双端口与实时状态（3周）
- - [ ] µWebSockets集成：Agent状态实时推送（思考、工具调用）
- - [ ] 前端Agent UI：思考过程可视化、Tool卡片、状态指示器
- - [ ] ReAct可视化：前端展示循环迭代过程（树状图）
 Phase 3: 高级工具与多Agent（4周）
- - [ ] AnalyzeTool：代码AST分析、复杂度计算、依赖图生成
- - [ ] SearchTool：本地代码向量检索（SQLite-VSS集成）
- - [ ] CoordinatorAgent：任务拆解、子Agent协调
- - [ ] Agent切换：Session内动态切换Agent类型（保留上下文）
 Phase 4: 记忆与上下文（3周）
- - [ ] 工作记忆：滑动窗口管理、Token超限摘要压缩
- - [ ] 长期记忆：跨Session向量存储、相似任务推荐
- - [ ] 执行轨迹持久化：保存完整ReAct过程供用户复盘
 Phase 5: 安全与优化（4周）
- - [ ] Tool权限细化：细粒度ACL（如FileTool只读/读写模式）
- - [ ] 沙箱增强：网络隔离、资源限制、命令白名单动态更新
- - [ ] LLM缓存：相似Prompt结果缓存，减少llama.cpp调用
- - [ ] 并行Tool执行：独立Tools并行调用（如同时读多个文件）
 Phase 6: 生态与扩展（持续）
- - [ ] Tool Marketplace：动态加载外部Tool（.so插件）
- - [ ] Agent Template：用户自定义Agent角色、工具集
- - [ ] Multi-Session Agent：跨Session协作（代码库级别Agent）

    auto body = json::parse(req.body);
    std::string agent_type = body.value("agent_type", "coder");
    
    auto session = session_manager.CreateSession(body["name"], agent_type);
    res.set_content(session->ToJson().dump(), "application/json");
});

// 与Agent对话（核心入口）
svr.post("/api/session/:id/agent/chat", [](const httplib::Request& req, httplib::Response& res) {
    auto session = session_manager.GetSession(req.path_params.at("id"));
    auto agent = agent_manager.GetAgent(session->agent_id);
    
    auto body = json::parse(req.body);
    std::string user_input = body["message"];
    
    // SSE流式返回
    res.set_header("Content-Type", "text/event-stream");
    res.set_chunked_content_provider(
        "text/event-stream",
        [agent, user_input](size_t offset, httplib::DataSink& sink) {
            // 注册回调：Token流式输出
            agent->ProcessUserInput(user_input,
                [&sink](const std::string& token) {
                    json chunk = {{"type": "token", "content": token}};
                    sink.write(("data: " + chunk.dump() + "\n\n").c_str());
                },
                [&sink](const json& final_result) {
                    json chunk = {{"type": "final", "result": final_result}};
                    sink.write(("data: " + chunk.dump() + "\n\n").c_str());
                }
            );
            return true;
        }
    );
});

// µWebSockets 消息协议定义

interface AgentStateMessage {
    type: 'agent_state';
    session_id: string;
    agent_id: string;
    state: 'IDLE' | 'PLANNING' | 'ACTION' | 'OBSERVATION' | 'FINAL_ANSWER';
    current_iteration: number;
    max_iterations: number;
    timestamp: number;
}

interface ThoughtMessage {
    type: 'thought';
    session_id: string;
    content: string;  // AI的思考过程
    step_number: number;
}

interface ToolCallMessage {
    type: 'tool_call';
    session_id: string;
    tool_call_id: string;
    tool_name: string;
    parameters: Record<string, any>;
    timestamp: number;
}

interface ToolResultMessage {
    type: 'tool_result';
    session_id: string;
    tool_call_id: string;
    status: 'success' | 'error' | 'timeout';
    result: any;  // 执行结果或错误信息
    execution_time_ms: number;
}
10. 与原版架构对比
维度	原版 (Session-Only)	新版 (Agentic)
交互模式	直接透传（用户↔LLM）	智能体编排（用户↔Agent↔[Tools+LLM]）
状态管理	简单的对话历史	复杂的状态机（ReAct循环）
工具使用	无	结构化Tool调用（File/Terminal/Analyze）
执行能力	仅回答	可自主执行代码、修改文件、运行命令
多步任务	单轮/多轮对话	自主规划-执行-观察循环
前端展示	聊天窗口	聊天 + Agent思维导图 + Tool执行面板
记忆深度	Session内短期	工作记忆+长期向量记忆
专业分工	通用助手	CoderAgent/TerminalAgent/CoordinatorAgent
